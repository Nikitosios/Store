Установка Arch Linux + xfce4
loadkeys ru                                             /* загрузка русского языка */
setfont cyr-sun16                                       /* установка шрифта с поддержкой кириллицы */
dhcpcd                                                  /* подключение проводного интернета */
wifi-menu                                               /* или wi-fi */
timedatectl set-ntp true                                /* синхронизация системных часов с сетью */
fdisk -l                                                /* смотрим какие есть разделы */
fdisk /dev/sda                                          /* разбиваем диск */
mkfs.[FS] -L [label]                                    /* форматируем разделы */
mount ... /mnt/...                                      /* монтируем нужные разделы в нужные места */
vim /etc/pacman.d/mirrorlist                            /* перемещаем ближайшие зеркала вверх */
pacstrap /mnt base base-devel                           /* ставим базовую систему */
genfstab -p /mnt >> /mnt/etc/fstab                      /* генерируем fstab */
arch-chroot /mnt                                        /* чрутимся в новоиспеченный арч */
echo [Имя компа] > /etc/hostname                        /* устанавливаем имя компьютера */
ln -s/usr/share/zoneinfo/[Регион]/[Город]               /* ставим нужный часовой пояс */
vi /etc/locale.gen                                      /* раскомментируем нужные локали */
/* А именно: ru_RU.UTF-8, ru_UA.UTF-8, en_US.UTF-8 */
locale-gen                                              /* обновим локаль системы согласно новым правилам */
echo 'LANG=ru_UA.UTF-8' > /etc/locale.conf              /* ставим стандартную локаль */
echo 'KEYMAP=ru' > /etc/vconsole.conf                   /* добавляем русский в список доступных языков */
echo 'FONT=cyr-sun16' >> /etc/vconsole.conf             /* врубаем шрифт с поддержкой кириллицы */
mkinitcpio -p linux                                     /* чето типо генерации initrd */
passwd                                                  /* меняем пароль рута */
pacman -S grub                                          /* ставим grub утилиты */
pacman -S os-prober mtools fuse                         /* ставим доп утилиты для граба (я так понял) */
grub-install /dev/sda                                   /* ставим grub */
grub-mkconfig -o /boot/grub/grub.cfg                    /* конфигурируем grub */
pacman -S dialog wpa_supplicant                         /* для wi-fi */
exit                                                    /* вылазим отсюда */
umount /mnt/*                                           /* приберемся за собой */
reboot                                                  /* ребутимся, вынимаем флешку */
/* логинимся в рута */
dhcpcd                                                  /* подключаем проводной интернет */
wifi-menu                                               /* или wi-fi */
useradd -m -s /bin/bash [ЮЗЕР]                          /* добавляем нужных юзеров */
passwd [ЮЗЕР]                                           /* ставим пароли на нужных юзеров */
/* В файле /etc/sudoers раскомментируем строку '%wheel ALL=(ALL) ALL' (настройка sudo) */
/* для работы 32-битных приложений нужно раскомментировать репозиторий multilib в /etc/pacman.conf */
// [multilib]
// Include = /etc/pacman.d/mirrorlist
pacman -Syy                                             /* обновляем заголовки репозиториев */
pacman -S xorg-server xorg-drivers                      /* ставим иксы и дрова */
pacman -S networkmanager network-manager-applet ppp     /* ставим менеджер сети */
systemctl enable NetworkManager                         /* подрубаем тырнет */
pacman -S xf86-video-ati lib32-ati-dri                  /* ставим дрова для видюхи AMD */
pacman -Syy                                             /* обновляем заголовки реп */
/* добавляем в файл /etc/pacman.conf строки: */
// [archlinuxfr]
// SigLevel=Never
// Server=http://repo.archlinux.fr/$arch
pacman -Syu                                             /* обновляемся */
pacman -Sy yaourt                                       /* ставим AUR манагер */
pacman -S xfce4 xfce4-goodies lxdm                      /* ставим xfce */
reboot                                                  /* ребутаемся */
vi /etc/lxdm/lxdm.conf                                  /* раскомментировать строку autologin= и дописать юзера */
yaourt -S vertex-themes lxdm-themes                     /* ставим доп. темы */
vi /etc/lxdm/lxdm.conf                                  /* заменить theme=Industrial на theme=ArchlinuxTop */
/* настройки -> клавиатура -> комбинации клавиш */
// Caps Lock    переключение раскладки
// Ctrl+Alt+Del xfce4-taskmanager
// Ctrl+Atl+T   xfce4-terminal
// Print Screen xfce4-screenshooter -f
/* доп. инфа на сайте https://wiki.archlinux.org/index.php/Xfce_(Русский) */
/* не забудь поиграться с темами! */
pacman -S [тут твоя туча программ]                      /* ставим проги */
yaourt -S timeshift                                     /* делаем бэкап системы */
yaourt -Syua                                            /* обновляем тупо всё */
